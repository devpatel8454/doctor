app.post("/login", async (req, res) => {
    try {
        if (!req.body.email || !req.body.password) {
            return res
                .status(400)
                .json({ message: "Email and password required" });
        }

        const user = await Developer.findOne({ email: req.body.email });

        if (!user) {
            return res.status(400).json({ message: "Email does not exist" });
        }

        const match = await bcrypt.compare(req.body.password, user.password);
        if (!match) {
            return res.status(400).json({ message: "Password is incorrect" });
        }

        req.session.userId = user._id;
        res.redirect("home")
    } catch (error) {
        res.status(500).json({ message: "Error in login", error: error.message });
    }
});


app.get("/home", async (req, res) => {
    try {
        if (!req.session.userId) {
            return res.send("Go login first");
        }

        const user = await Developer.findById(req.session.userId).populate("degree");

        if (!user) return res.send("User not found");

        const colleagues = await Developer.find({
            city: user.city,
            _id: { $ne: user._id },
        });

        res.render("home", { user, colleagues }); 
    } catch (error) {
        res.status(500).send("Error loading home page");
    }
});


app.get("/logout", (req, res) => {
    req.session.destroy();
    res.redirect("login");
});

app.listen(3030, () => {
    console.log("Server running on port 3030");
});

  const [data, setData] = useState(null);

  useEffect(() => {
    fetch('http://localhost:3030/api/home', {credentials: 'include'})
    .then(res => res.json())

  },[]);

  if(!data) return <h1>Loading...</h1>

  return (
    <>

  )
